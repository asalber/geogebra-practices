% \subsection{Integral de Riemann}

% Se llama \emph{partición} de un intervalo $[a,b]\subset\mathbb{R}$,
% a una colección finita de puntos del intervalo,
% $P=\{x_{0},x_{1},...,x_{n}\}$,  tales que
% $x_{0}=a<x_{1}<...<x_{n}=b$, con lo que el intervalo $[a,b]$ queda
% dividido en $n$ subintervalos $[x_{i},x_{i+1}]$, $i=0,...,n-1$.

% Dada una función $f:[a,b]\rightarrow\mathbb{R}$ acotada y una
% partición $P=\{x_{0},x_{1},...,x_{n}\}$ de $[a,b]$, se llama
% \emph{suma inferior} de $f$ en relación a $P$, y se designa por
% $L(P,f)$, a:

% \[
% \ L(P,f)=\sum_{i=1}^{n} m_{i}(x_{i}-x_{i-1})
% \]
% donde $  m_{i}=\inf\{f(x):x_{i-1}\leq x \leq x_{i}\}$.

% Análogamente se llama \emph{suma superior} de $f$ en relación a $P$,
% y se designa por $U(P,f)$, a:

% \[
% \ U(P,f)=\sum_{i=1}^{n} M_{i}(x_{i}-x_{i-1})
% \]
% donde $ M_{i}=\sup\{f(x):x_{i-1}\leq x \leq x_{i}\}$.

% La \emph{suma inferior} y la \emph{suma superior} así definidas
% representan las sumas de las áreas de los rectángulos que tienen por
% bases los subintervalos de la partición, y por alturas los valores
% mínimo y máximo respectivamente de la función $f$ en los
% subintervalos considerados, tal y como se muestra en la
% figura~\ref{g:sumassupinf}. Así, los valores de $L(P,f)$ y $U(P,f)$
% serán siempre menores y mayores respectivamente, que el área
% encerrada por la función $f$ y el eje de abscisas en el intervalo
% $[a,b]$.

\section{Solved exercises}

\begin{enumerate}[leftmargin=*]
\item Compute the following integrals:
      \begin{enumerate}
      \item $ \int{x^2 \log x\,dx}$
            \begin{indication}
            Enter the command \command{Integral(x\^{}2log(x))} in the \field{Input Bar} of the \field{CAS View}.
            \end{indication}

      \item $\displaystyle \int \frac{\log(\log x)}{x}\,dx$
            \begin{indication}
            Enter the command \command{Integral(log(log(x))/x)} in the \field{Input Bar} of the \field{CAS View}.
            \end{indication}

      \item $\displaystyle \int \frac{5x^{2}+4x+1}{x^{5}-2x^{4}+2x^{3}-2x^{2}+x}\,dx$
            \begin{indication}
            Enter the command \command{Integral((5x\^{}2+4x+1)/(x\^{}5-2x\^{}4+2x\^{}3-2x\^{}2+x))} in the \field{Input Bar} of the \field{CAS View}.
            \end{indication}

      \item $\displaystyle \int \frac{6x+5}{(x^{2}+x+1)^{2}}\,dx$
            \begin{indication}
            Enter the command \command{Integral((6x+5)/((x\^{}2+x+1)\^{}2))} in the \field{Input Bar} of the \field{CAS View}.
            \end{indication}
      \end{enumerate}


\item Compute the following definite integrals and plot them:
      \begin{enumerate}
      \item $\displaystyle \int_{-\frac{1}{2}}^0 \frac{x^{3}}{x^{2}+x+1}\,dx$
            \begin{indication}
            \begin{enumerate}
            \item Enter the function \command{f(x):=x\^{}3/(x\^{}2+x+1)} in the \field{Input Bar} of the \field{CAS View} and activate the \field{Graphics View}.
            \item Enter the command \command{Integral(f(x), -1/2, 0)} and click the button \button{Numeric Evaluation}.
            \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
            \end{enumerate}
            \end{indication}

      \item $\displaystyle \int_2^4 \frac{\sqrt{16-x^{2}}}{x}\,dx$
            \begin{indication}
            \begin{enumerate}
            \item Enter the function \command{g(x):=sqrt(16-x\^{}2)/x} in the \field{Input Bar} of the \field{CAS View}.
            \item Enter the command \command{Integral(g(x), 2, 4)} and click the button \button{Numeric Evaluation}.
            \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
            \end{enumerate}
            \end{indication}

      \item $\displaystyle \int_0^{\frac{\pi}{2}} \frac{dx}{3+\cos(2x)}$
            \begin{indication}
            \begin{enumerate}
            \item Enter the function \command{h(x):=1/(3+cos(2x))} in the \field{Input Bar} of the \field{CAS View}.
            \item Enter the command \command{Integral(h(x), 0, pi/2)} and click the button \button{Numeric Evaluation}.
            \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
            \end{enumerate}
            \end{indication}
      \end{enumerate}

\item Compute the following improper integral $\int_2^{\infty} x^2e^{-x}\,dx$ and plot it.
      \begin{indication}
      \begin{enumerate}
      \item Enter the function \command{f(x):=x\^{}2 exp(-x)} in the \field{Input Bar} of the \field{CAS View} and activate the \field{Graphics View}.
      \item Enter the command \command{Integral(f(x), 2, inf)} and click the button \button{Numeric Evaluation}.
      \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
      \end{enumerate}
      \end{indication}


\item Plot the graph of the parabola $y=x^2-7x+6$ and compute the area between the parabola and the horizontal axis, limited by the lines $x=2$ and $x=7$.
      \begin{indication}
      \begin{enumerate}
      \item Enter the function \command{f(x):=x\^{}2-7x+6} in the \field{Input Bar} of the \field{CAS View} and activate the \field{Graphics View}.
      \item Since $f$ takes positive and negative values in the interval $(2,7)$, to get the area between $f$ and the horizontal axis in this interval you have to compute the integral of the absolute value of $f$. For it, enter the command \command{Integral(abs(f(x)), 2, 7)} and click the button \button{Numeric Evaluation}.
      \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
      \end{enumerate}
      \end{indication}


\item Compute and plot the area between the graphs of the functions $\sin x$ and $\cos x$ in the interval $[0,2\pi]$.
      \begin{indication}
      \begin{enumerate}
      \item Enter the function \command{f(x):=sin(x)} in the \field{Input Bar} of the \field{CAS View} and activate the \field{Graphics View}.
      \item Enter the function \command{g(x):=cos(x)} in the \field{Input Bar}.
      \item Enter the command \command{Integral(abs(f(x)-g(x)), 0, 2pi)} in the \field{Input Bar} and click the button \button{Numeric Evaluation}.
      \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
      \item Another alternative is using the command \command{Integral(Max(f(x),g(x)), Min(f(x),g(x)), 0, 2pi)}.
      \end{enumerate}
      \end{indication}


\item Plot the region of the first quadrant limited by the function $f(x)=\sin x+2$, the line $x=2\pi$ and the horizontal axis, and compute the volume of the solid of revolution generated rotating the area around the horizontal axis.
      \begin{indication}
      \begin{enumerate}
      \item Enter the function \command{f(x):=sin(x)+2} in the \field{Input Bar} of the \field{CAS View} and activate the \field{Graphics View}.
      \item To compute the area of the region enter the command \command{Integral(f(x), 0, 2pi)} in the \field{Input Bar} and click the button \button{Numeric Evaluation}.
      \item To plot the region that covers the definite integral, click the circle to the left of the previous expression.
      \item To compute the volume of the solid of revolution enter the command \command{Integral(pif(x)\^{}2, x, 0, 2pi)}.
      \end{enumerate}
      \end{indication}

\end{enumerate}
